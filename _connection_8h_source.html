88 
89  asio::async_write(socket, asio::buffer(&msg.header, sizeof(msg.header)),
90  [this](std::error_code ec, size_t size) {
91  if(ec)

121 template <typename T>
122 void Connection::readHeader()
123 {
124  Message<T>& msg = *static_cast<Message<T>*>(buffer);
125  async_read(socket, asio::buffer(&msg.header, sizeof(Message<T>::header)),
126  [&, this](std::error_code ec, size_t size) {
127  if(ec)
128  {
129  if(onError)
130  onError(ec);
131  CLEM_LOG_ERROR("networking", ec.message());
132  return;
133  }
134 
135  if(msg.header.size == 0)
136  {
137  if(onMessage)
138  onMessage();
139  readHeader<T>();
140  return;
141  }
142 
143  msg.body.resize(msg.header.size);
144  readBody<T>();
145  });
146 }
